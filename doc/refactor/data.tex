\subsection{Calculation results}
\begin{tikzpicture}
    \begin{abstractclass}[text width=30em]{CavitiesBase}{0, 0}
        \attribute{+ timestamp : datetime}
        \attribute{+ number : Integer}
        \attribute{+ volumes : Array(Float)}
        \attribute{+ surface\_areas : Array(Float)}
        \attribute{+ triangles : list(Array(Float))}

        \operation{+ CavitiesBase(timestamp : datetime, volumes : Array(Float), surface\_areas : Array(Float), triangles : list(Array(Float)))}
        \operation{+ CavitiesBase(h5group : HDF5Group)}
        \operation{+ tohdf(h5group : HDF5Group)}
    \end{abstractclass}

    \begin{class}[text width=30em]{Domains}{-1, -8}
        \inherit{CavitiesBase}
        
        \attribute{+ centers : Array(Float)}

        \operation{+ Domains(timestamp : datetime, volumes : Array(Float), surface\_areas : Array(Float), triangles : list(Array(Float)), centers : Array(Float))}
        \operation{+ Domains(h5group : HDF5Group)}
        \operation{+ tohdf(h5group : HDF5Group)}
    \end{class}

    \begin{class}[text width=30em]{Cavities}{1, -13}
        \inherit{CavitiesBase}

        \attribute{+ multicavities : list(Array(Integer))}

        \operation{+ Cavities(timestamp : datetime, volumes : Array(Float), surface\_areas : Array(Float), triangles : list(Array(Float)), multicavities : list(Array(Integer)))}
        \operation{+ Cavities(h5group : HDF5Group)}
        \operation{+ tohdf(h5group : HDF5Group)}
    \end{class}
\end{tikzpicture}
